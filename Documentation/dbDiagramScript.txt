Table UserMstr as UM{
  UserId int [pk]
  FirstName varchar(16)
  LastName varchar(16)
  UserName varchar(16)
  Password varchar(16)
  Company varchar(256)
  Street varchar(256)
  AddressId int
  Email varchar(32)
  PhNo varchar(10)
  IsVerified boolean
  Photo varchar(128)
  Role tinyint
  CreatedDate datetime
  ModifiedDate datetime
}
